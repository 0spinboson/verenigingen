def test_portal_access_pattern(self):
    """Standard pattern for testing portal access"""
    frappe.set_user(self.test_user_email)
    
    context = {}
    get_context(context)
    
    # Verify access granted
    self.assertIsNotNone(context.get("volunteer"))
